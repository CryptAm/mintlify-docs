## `Call`

```ts
type Call = { to: Hex; data?: Hex; value?: bigint };
```

## `Calls`

```ts
type Calls = Call[] | Promise<Call[]> | (() => Promise<Call[]>);
```

## `Contracts`

```ts
type Contracts =
  | ContractFunctionParameters[]
  | Promise<ContractFunctionParameters[]>
  | (() => Promise<ContractFunctionParameters[]>);
```

## `LifecycleStatus`

```ts
type LifecycleStatus =
  | {
      statusName: 'init';
      statusData: null;
    }
  | {
      statusName: 'error';
      statusData: TransactionError;
    }
  | {
      statusName: 'transactionIdle'; // начальный статус до выполнения функции мутаци
      statusData: null;
    }
  | {
      statusName: 'buildingTransaction'; // разрешение промисов calls или contracts
      statusData: null;
    }
  | {
      statusName: 'transactionPending'; // если мутация в данный момент выполняется
      statusData: null;
    }
  | {
      statusName: 'transactionLegacyExecuted';
      statusData: {
        transactionHashList: Address[];
      };
    }
  | {
      statusName: 'success'; // если последняя попытка мутации была успешной
      statusData: {
        transactionReceipts: TransactionReceipt[];
      };
    };
```

## `TransactionError`

```ts
type TransactionError = {
  code: string; // Код ошибки, представляющий тип ошибки транзакции.
  error: string; // Сообщение об ошибке с деталями об ошибке транзакции.
  message: string; // Сообщение об ошибке с деталями об ошибке транзакции.

```

## `TransactionReact`

```ts
type TransactionReact = {
  calls?: Calls | Contracts | (Call | ContractFunctionParameters)[]; // Массив вызовов (calls) для выполнения в транзакции.
  isSponsored?: boolean; // Спонсируется ли транзакция.
  /** @deprecated Используйте `isSponsored` вместо этого. */
  capabilities?: WalletCapabilities; // Возможности, поддерживаемые кошельком (например, paymasters, сессионные ключи и т.д.).
  chainId?: number; // Chain ID для транзакции.
  className?: string; // Необязательное имя CSS-класса для стилизации компонента.
  contracts?: Calls | Contracts | (Call | ContractFunctionParameters)[]; // Массив контрактов (contracts) для выполнения в транзакции.
  onError?: (e: TransactionError) => void; // Необязательная функция обратного вызова для обработки ошибок транзакции.
  onStatus?: (lifecycleStatus: LifecycleStatus) => void; // Необязательная функция обратного вызова, предоставляющая доступ к состоянию жизненного цикла компонента
  onSuccess?: (response: TransactionResponse) => void; // Необязательная функция обратного вызова, предоставляющая доступ к квитанциям о транзакции
} & (
  | {
      children: ReactNode;
      /** Необязательный проп для отключения кнопки отправки. Доступен только когда children не предоставлены. */
      disabled?: never;
    }
  | {
      children?: never;
      /** Необязательный проп для отключения кнопки отправки. Доступен только когда children не предоставлены. */
      disabled?: boolean;
    }
);
```

## `TransactionDefaultReact`

```ts
export type TransactionDefaultReact = {
```
