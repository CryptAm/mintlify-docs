## `ConnectWalletReact`

```ts
type ConnectWalletReact = {
  children?: React.ReactNode; // Дочерние элементы можно использовать для отображения пользовательского контента, когда кошелек подключен.
  className?: string; // Необязательное переопределение className для элемента кнопки
  text?: string; // Необязательное переопределение текста для кнопки. Примечание: предпочтительнее использовать проп `disconnectedLabel`, так как эта опция будет удалена в будущей версии.
  disconnectedLabel?: React.ReactNode; // Необязательное переопределение текста для кнопки.
  onConnect?: () => void; // Необязательная функция обратного вызова, которая вызывается при подключении кошелька. Может использоваться для запуска запросов SIWE или других действий.
};
```

## `WalletContextType`

```ts
type WalletContextType = {
  address?: Address | null; // Адрес Ethereum, для которого нужно получить аватар и имя.
  chain?: Chain; // Необязательная цепь для разрешения домена
  isConnectModalOpen: boolean;
  setIsConnectModalOpen: Dispatch<SetStateAction<boolean>>;
  isSubComponentOpen: boolean;
  setIsSubComponentOpen: Dispatch<SetStateAction<boolean>>;
  isSubComponentClosing: boolean;
  setIsSubComponentClosing: Dispatch<SetStateAction<boolean>>;
  handleClose: () => void;
  connectRef: React.RefObject<HTMLDivElement>;
  showSubComponentAbove: boolean;
  alignSubComponentRight: boolean;
  activeFeature: WalletAdvancedFeature | null;
  setActiveFeature: Dispatch<SetStateAction<WalletAdvancedFeature | null>>;
  isActiveFeatureClosing: boolean;
  setIsActiveFeatureClosing: Dispatch<SetStateAction<boolean>>;
  tokenBalances: PortfolioTokenWithFiatValue[] | undefined;
  portfolioFiatValue: number | undefined;
  isFetchingPortfolioData: boolean;
  portfolioDataUpdatedAt: number | undefined;
  refetchPortfolioData: () => Promise<QueryObserverResult<Portfolio, Error>>;
  animations: {
    container: string;
    content: string;
  };
};
```

## `WalletDropdownFundLinkReact`

```ts
export type WalletDropdownFundLinkReact = {
  className?: string; // Необязательное переопределение className для элемента
  icon?: ReactNode; // Необязательное переопределение иконки
  openIn?: 'popup' | 'tab'; // Открыть поток пополнения во вкладке или всплывающем окне
  popupSize?: 'sm' | 'md' | 'lg'; // Размер всплывающего окна, если `openIn` установлен в `popup`
  rel?: string; // Определяет связь между текущим документом и связанным документом
  target?: string; // Где открывать цель, если `openIn` установлен в tab
  text?: string; // Необязательное переопределение текста
};
```
