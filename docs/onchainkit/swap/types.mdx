## `LifecycleStatus`

```ts
type LifecycleStatusDataShared = {
  isMissingRequiredField: boolean;
  maxSlippage: number;
};

type LifecycleStatus =
  | {
      statusName: 'init';
      statusData: LifecycleStatusDataShared;
    }
  | {
      statusName: 'error';
      statusData: SwapError & LifecycleStatusDataShared;
    }
  | {
      statusName: 'amountChange';
      statusData: {
        amountFrom: string;
        amountTo: string;
        tokenFrom?: Token;
        tokenTo?: Token;
      } & LifecycleStatusDataShared;
    }
  | {
      statusName: 'slippageChange';
      statusData: LifecycleStatusDataShared;
    }
  | {
      statusName: 'transactionPending';
      statusData: LifecycleStatusDataShared;
    }
  | {
      statusName: 'transactionApproved';
      statusData: {
        transactionHash: Hex;
        transactionType: 'ERC20' | 'Permit2';
      } & LifecycleStatusDataShared;
    }
  | {
      statusName: 'success';
      statusData: {
        transactionReceipt: TransactionReceipt;
      } & LifecycleStatusDataShared;
    };
```

## `SwapQuote`

```ts
type SwapQuote = {
  amountReference: string; // Референсное количество для котировки
  from: Token; // Исходный токен для обмена
  fromAmount: string; // Количество исходного токена
  fromAmountUSD: string; // Стоимость в USD исходного токена
  hasHighPriceImpact: boolean; // Имеет ли обмен высокое влияние на цену
  priceImpact: string; // Влияние на цену при обмене
  slippage: string; // Проскальзывание при обмене

```

## `SwapReact`

```ts
type SwapReact = {
  children: ReactNode;
  className?: string; // Необязательное переопределение className для верхнего div-элемента.
  config?: {
    maxSlippage: number; // Максимально допустимое проскальзывание (slippage) для обмена (например, 3 для 3%). Это процент, а не базисные пункты;
  };
  experimental?: {
    /**
     * Использовать ли агрегатор DEX.
     * true - 0x Aggregator
     * false - Uniswap V3
     * @default false
     */
    useAggregator: boolean;
  };
  isSponsored?: boolean; // Необязательная настройка для спонсирования обменов с помощью Paymaster. (по умолчанию: false)
  onError?: (error: SwapError) => void; // Необязательная функция обратного вызова для обработки ошибок внутри провайдера.
  onStatus?: (lifecycleStatus: LifecycleStatus) => void; // Необязательная функция обратного вызова, предоставляющая доступ к состоянию жизненного цикла компонента
  onSuccess?: (transactionReceipt: TransactionReceipt) => void; // Необязательная функция обратного вызова, предоставляющая доступ к квитанции о транзакции (receipt)
  title?: string; // Заголовок для компонента Swap. (по умолчанию: "Swap")
};
```

## `Transaction`

```ts
type Transaction = {
  chainId: number; // Идентификатор цепи (Chain ID)
  data: Hex; // Данные для транзакции (hex-строка)
  gas: bigint; // Лимит газа
  maxFeePerGas?: bigint | undefined; // Максимальная цена за газ
  maxPriorityFeePerGas?: bigint | undefined; // Максимальная приоритетная цена за газ
  nonce?: number; // nonce для транзакции
  to: Address; // Адрес получателя
  value: bigint; // Значение транзакции
};
```
