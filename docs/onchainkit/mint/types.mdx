## `LifecycleStatus`

```ts
export type LifecycleStatus =
  | {
      statusName: 'init';
      statusData: null;
    }
  | {
      statusName: 'error';
      statusData: NFTError;
    }
  | {
      statusName: 'mediaLoading';
      statusData: {
        mediaType: MediaType;
        mediaUrl: string;
      };
    }
  | {
      statusName: 'mediaLoaded';
      statusData: null;
    }
  | {
      statusName: 'transactionPending';
      statusData: null;
    }
  | {
      statusName: 'transactionLegacyExecuted';
      statusData: {
        transactionHashList: Address[];
      };
    }
  | {
      // Состояние успеха для NFTCard означает загруженный медиафайл
      // Состояние успеха для NFTMintCard означает успешный минт
      statusName: 'success';
      statusData: {
        transactionReceipts?: TransactionReceipt[];
      };
    };
```

## `NFTCardReact`

```ts
type NFTCardReact = {
  children: React.ReactNode;

  // Необязательное переопределение className для верхнего div-элемента.
  className?: string;

  // Адрес контракта NFT (hex-строка)
  contractAddress: Hex;

   // Обязательный идентификатор токена (Token ID) NFT
  tokenId: string;

  // Необязательный хук для переопределения хука useNftData по умолчанию
  useNFTData?: UseNFTData;

   // Необязательная функция обратного вызова для обработки ошибок внутри провайдера.
  onError?: (error: NFTError) => void;

  // Необязательная функция обратного вызова, предоставляющая доступ к состоянию жизненного цикла компонента
  onStatus?: (lifecycleStatus: LifecycleStatus) => void;

  // Для карточки (card) transactionReceipt не передается
  onSuccess?: (transactionReceipt?: TransactionReceipt) => void;
};
```

## `NFTMintCardReact`
```ts
type NFTMintCardReact = {
  children: ReactNode;

  // Необязательное переопределение className для верхнего div-элемента.
  className?: string;

  // Адрес контракта NFT (hex-строка)
  contractAddress: Hex;

  // Идентификатор токена (Token ID) требуется только для ERC1155
  tokenId?: string;

  // Необязательный флаг, определяющий, спонсируется ли минт через paymaster
  isSponsored?: boolean;

  // Необязательный хук для переопределения хука useNFTData по умолчанию
  useNFTData?: UseNFTData;

  // Необязательная функция для переопределения функции по умолчанию, которая создает транзакцию минта
  buildMintTransaction?: BuildMintTransaction;

  // Необязательная функция обратного вызова для обработки ошибок внутри провайдера.
  onError?: (error: NFTError) => void;

  // Необязательная функция обратного вызова, предоставляющая доступ к состоянию жизненного цикла компонента
  onStatus?: (lifecycleStatus: LifecycleStatus) => void;

  // MintCard передаст transactionReceipt
  onSuccess?: (transactionReceipt?: TransactionReceipt) => void;
};
```

## `NFTData`
```ts
type NFTData = {
  // компоненты карточки (card)
  name?: string; // требуется для NFTTitle и NFTCollectionTitle
  description?: string; // в настоящее время не используется
  imageUrl?: string; // требуется для NFTMedia
  animationUrl?: string; // требуется для NFTMedia (типы audio и video)
  /* Поддерживаемые mimeTypes:
   * изображение = image/*
   * видео = video/*
   * аудио = audio/* | application/*
   */
  lastSoldPrice?: NFTPrice; // требуется для NFTLastSoldPrice
  mimeType?: string; // требуется для NFTMedia (возвращается к изображению по умолчанию)

  // компоненты минта
  ownerAddress?: `0x${string}`; // требуется для NFTOwner
  contractType?: ContractType;
  mintDate?: Date; // требуется для NFTMintDate
  price?: NFTPrice; // требуется для NFTAssetCost, NftTotalCost
  mintFee?: NFTPrice; // требуется для NFTTotalCost
  creatorAddress?: Hex; // требуется для NFTCreator
  maxMintsPerWallet?: number; // требуется для NFTMintButton
  isEligibleToMint?: boolean; // требуется для NFTMintButton
  totalOwners?: string; // требуется для NFTMinters
  recentOwners?: Address[]; // требуется для NFTMinters
  network?: string; // требуется для реализации BuildMintTransaction по умолчанию
};
```
