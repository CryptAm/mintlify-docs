import { useAddFrame } from '@coinbase/onchainkit/minikit';

      } else {
        console.log('User cancelled or frame already saved');
      }
    } catch (error) {
      console.error('Failed to save frame:', error);
    } finally {
      setIsAdding(false);
    }
  };

  return (

      disabled={isAdding}

  });
}
