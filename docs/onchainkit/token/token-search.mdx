`<TokenSearch />` - это компонент поиска с необязательной задержкой устранения дребезга (debounce).

Если вы хотите обрабатывать задержку устранения дребезга вне этого компонента, установите `delayMs` в `0`.

## Использование

Используйте [`getTokens`](/onchainkit/api/get-tokens) и `<TokenSearch />` вместе для поиска [`Token`](/onchainkit/token/types#token).

```tsx
// @noErrors: 7006 2322 1128 - 'value' implicitly has any type, Type GetTokensResponse is not assignable to type Token[]
import { useCallback } from 'react';
import { base } from 'viem/chains';
// ---cut-before---
import { OnchainKitProvider } from '@coinbase/onchainkit';
import { getTokens } from '@coinbase/onchainkit/api'; // [!code focus]
import { TokenSearch } from '@coinbase/onchainkit/token'; // [!code focus]
import type { Token } from '@coinbase/onchainkit/token'; // [!code focus]

...

// пример асинхронного обработчика onChange
const handleChange = useCallback((value: string) => {
  async function getData(value) {
    const tokens: Token[] = await getTokens({ search: value }); // [!code focus]
    // Сделайте что-нибудь с полученными токенами здесь
  }
  getData(value);
}, []);

...

<OnchainKitProvider
  chain={base}
  apiKey="YOUR_API_KEY"
>
  <TokenSearch onChange={handleChange} delayMs={200} />
</OnchainKitProvider>
```
